    if archive_clean['p1_dog'] == True:
        dog_type.append(archive_clean['p1'])

    elif archive_clean['p2_dog'] == True:
        dog_type.append(archive_clean['p2'])
        confidence_list.append(archive_clean['p2_conf'])
